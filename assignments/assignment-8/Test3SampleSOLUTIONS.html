# Write Some Code #1:
infile = open('profits.txt', 'r')
item = infile.readline()
if item == '':
print('Error: file is empty')
else:
highest = float(item)
lowest = float(item)
sum = 0.0
count = 0
while item != '':
num = float(item)
sum += num
if num > highest:
highest = num
elif num < lowest:
lowest = num
count += 1
item = infile.readline()
average = sum / count
print('lowest:', lowest)
print('highest: ' + str(highest))
print('average: %.2f' % average)
infile.close()

# Write Some Code #3:
def reverse(s):
result = ''
i = len(s)
while i > 0:
i -= 1
result += s[i]
return result

 

# Write Some Code #4:
def merge(a, b):
shorterLength = min(len(a), len(b))
result = ''
for i in range(shorterLength):
result += a[i] + b[i]
result += a[shorterLength:] # adds anything left in 'a'
result += b[shorterLength:] # adds anything left in 'b'
return result
